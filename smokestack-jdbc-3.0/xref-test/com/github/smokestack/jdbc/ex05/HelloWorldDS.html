10  import java.sql.Statement;

12  import javax.naming.InitialContext;
13  import javax.sql.DataSource;
14  
15  /**
16   * A very simple class that shows how to load the driver, create a database,
17   * create a table, and insert some data.
18   */
19  public class HelloWorldDS {
20  
21  	/**
22  	 * Called when ran from command line.
23  	 * 
24  	 * @param args ignored
25  	 */
26  	public static void main(String... args) throws Exception {
27  		InitialContext ic = new InitialContext();
28  		DataSource ds = (DataSource) ic.lookup("jdbc/dsName");
29  		Connection conn = ds.getConnection();
30  		Statement stat = conn.createStatement();
31  		ResultSet rs = null;
32  
33  		try{
34  			stat.execute("create table message(id int primary key, message varchar(255))");
35  			stat.execute("insert into message values(1, 'Hello World')");
36  			rs = stat.executeQuery("select * from message");
37  			while (rs.next()) {
38  				System.out.println(rs.getString("message"));
39  			}
40  		}
41  		finally{
42  			if(rs != null){
43  				rs.close();
44  			}
45  			if(stat != null){
46  				stat.close();
47  			}
48  			if(conn != null){
49  				conn.close();
50  			}
51  		}
52  	}
53  }
