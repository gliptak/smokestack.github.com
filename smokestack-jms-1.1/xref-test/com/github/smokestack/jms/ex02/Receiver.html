46  package com.github.smokestack.jms.ex02;
47  
48  import javax.naming.Context;
49  import javax.naming.InitialContext;
50  import javax.naming.NamingException;
51  
52  import javax.jms.JMSException;
53  import javax.jms.Message;
54  import javax.jms.Destination;
55  import javax.jms.Connection;
56  import javax.jms.ConnectionFactory;
57  import javax.jms.Session;
58  import javax.jms.MessageConsumer;
59  import javax.jms.TextMessage;
60  
61  
62  /**
63   * Synchronous <code>MessageConsumer</code> example.
64   *
65   * @author <a href="mailto:tma@netspace.net.au">Tim Anderson</a>
66   * @version $Revision: 1.2 $ $Date: 2005/11/18 03:28:01 $
67   */
68  public class Receiver {
69  
70      /**
71       * Main line.
72       *
73       * @param args command line arguments
74       */
75      public static void main(String[] args) {
76          Context context = null;
77          ConnectionFactory factory = null;
78          Connection connection = null;
79          String factoryName = "ConnectionFactory";
80          String destName = null;
81          Destination dest = null;
82          int count = 1;
83          Session session = null;
84          MessageConsumer receiver = null;
85  
86          if (args.length < 1 || args.length > 2) {
87              System.out.println("usage: Receiver <destination> [count]");
88              System.exit(1);
89          }
90  
91          destName = args[0];
92          if (args.length == 2) {
93              count = Integer.parseInt(args[1]);
94          }
95  
96          try {
97              // create the JNDI initial context
98              context = new InitialContext();
99  
100             // look up the ConnectionFactory
101             factory = (ConnectionFactory) context.lookup(factoryName);
102 
103             // look up the Destination
104             dest = (Destination) context.lookup(destName);
105 
106             // create the connection
107             connection = factory.createConnection();
108 
109             // create the session
110             session = connection.createSession(
111                 false, Session.AUTO_ACKNOWLEDGE);
112 
113             // create the receiver
114             receiver = session.createConsumer(dest);
115 
116             // start the connection, to enable message receipt
117             connection.start();
118 
119             for (int i = 0; i < count; ++i) {
120                 Message message = receiver.receive();
121                 if (message instanceof TextMessage) {
122                     TextMessage text = (TextMessage) message;
123                     System.out.println("Received: " + text.getText());
124                 } else if (message != null) {
125                     System.out.println("Received non text message");
126                 }
127             }
128         } catch (JMSException exception) {
129             exception.printStackTrace();
130         } catch (NamingException exception) {
131             exception.printStackTrace();
132         } finally {
133             // close the context
134             if (context != null) {
135                 try {
136                     context.close();
137                 } catch (NamingException exception) {
138                     exception.printStackTrace();
139                 }
140             }
141 
142             // close the connection
143             if (connection != null) {
144                 try {
145                     connection.close();
146                 } catch (JMSException exception) {
147                     exception.printStackTrace();
148                 }
149             }
150         }
151     }
152 }
